\documentclass[krantzl]{tex/krantz}

% Language rules.
\usepackage[english]{babel}

% Font encoding.
\usepackage[T1]{fontenc}

% Show page boundaries.
\usepackage[showframe]{geometry}

% Format code listings.
\usepackage{listings}
\lstset{basicstyle=\small\ttfamily,upquote=true}

% Include images.
\usepackage{graphicx}

% Make description items cross-referenceable
\def\namedlabel#1#2{\begingroup
    #2%
    \def\@currentlabel{#2}%
    \phantomsection\label{#1}\endgroup
}
\makeatother

% Glossary items and references.
\newcommand{\glossitem}[2]{\item[\hypertarget{#2}{#1}:]}
\newcommand{\glossref}[2]{\textbf{#1}}

% Use numbers for nested lists all the way down.
\usepackage{enumitem}
\setlist[enumerate,1]{label=\arabic*., ref=\arabic*}
\setlist[enumerate,2]{label=\arabic*., ref=\arabic*}
\setlist[enumerate,3]{label=\arabic*., ref=\arabic*}

% Remove indentation from footnotes
\usepackage[hang,flushmargin]{footmisc}

% Allow multi-page tables.
\usepackage{longtable}

% Cross-references.
\newcommand{\appref}[1]{Appendix~\ref{#1}}
\newcommand{\chapref}[1]{Chapter~\ref{#1}}

% Asides
\usepackage{framed}
\newenvironment{aside}{\begin{framed}\begin{quotation}}{\end{quotation}\end{framed}}
\newcommand{\asidetitle}[1]{\begin{center}{#1}\end{center}\vspace{\baselineskip}}

% URLs as footnotes.
% Always load 'hyperref' last (see link below for explanation).
% https://tex.stackexchange.com/questions/16268/warning-with-footnotes-namehfootnote-xx-has-been-referenced-but-does-not-exi
\newcommand{\hreffoot}[2]{{#1}\footnote{\href{#2}{#2}}}
